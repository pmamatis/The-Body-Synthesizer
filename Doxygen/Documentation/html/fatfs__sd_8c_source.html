95static void SD_PowerOn(void)
96{

98 uint32_t Count = 0x1FFF;
99
100
101 DESELECT();
102

104 {
105 SPI_TxByte(0xFF);

115 cmd_arg[3] = 0;

120 for (int i = 0; i < 6; i++)
121 {
122 SPI_TxByte(cmd_arg[i]);
123 }
124

128 Count--;
129 }
130
131 DESELECT();
132 SPI_TxByte(0XFF);

150/* 데이터 패킷 수신 */
151static bool SD_RxDataBlock(BYTE *buff, UINT btr)
152{
153 uint8_t token;
154
155 /* 100ms 타이머 */
156 Timer1 = 10;
157
158 /* 응답 대기 */
159 do
160 {
161 token = SPI_RxByte();
162 } while((token == 0xFF) && Timer1);
163
164 /* 0xFE 이외 Token 수신 시 에러 처리 */
165 if(token != 0xFE)
166 return FALSE;
167
168 /* 버퍼에 데이터 수신 */
169 do
170 {
171 SPI_RxBytePtr(buff++);
172 SPI_RxBytePtr(buff++);
173 } while(btr -= 2);
174
175 SPI_RxByte(); /* CRC 무시 */
176 SPI_RxByte();
177
178 return TRUE;
179}

233/* CMD 패킷 전송 */
234static BYTE SD_SendCmd(BYTE cmd, DWORD arg)
235{
236 uint8_t crc, res;
237
238 /* SD카드 대기 */
239 if (SD_ReadyWait() != 0xFF)
240 return 0xFF;
241
242 /* 명령 패킷 전송 */
243 SPI_TxByte(cmd); /* Command */
244 SPI_TxByte((BYTE) (arg >> 24)); /* Argument[31..24] */
245 SPI_TxByte((BYTE) (arg >> 16)); /* Argument[23..16] */
246 SPI_TxByte((BYTE) (arg >> 8)); /* Argument[15..8] */
247 SPI_TxByte((BYTE) arg); /* Argument[7..0] */
248
249 /* 명령별 CRC 준비 */
250 crc = 0;
251 if (cmd == CMD0)
252 crc = 0x95; /* CRC for CMD0(0) */
253
254 if (cmd == CMD8)
255 crc = 0x87; /* CRC for CMD8(0x1AA) */
256
257 /* CRC 전송 */
258 SPI_TxByte(crc);
259
260 /* CMD12 Stop Reading 명령인 경우에는 응답 바이트 하나를 버린다 */
261 if (cmd == CMD12)
262 SPI_RxByte();
263
264 /* 10회 내에 정상 데이터를 수신한다. */
265 uint8_t n = 10;
266 do
267 {
268 res = SPI_RxByte();
269 } while ((res & 0x80) && --n);
270
271 return res;
272}

279/* SD카드 초기화 */
280DSTATUS SD_disk_initialize(BYTE drv)
281{
282 uint8_t n, type, ocr[4];
283
284 /* 한종류의 드라이브만 지원 */
285 if(drv)
286 return STA_NOINIT;
287
288 /* SD카드 미삽입 */
289 if(Stat & STA_NODISK)
290 return Stat;
291
292 /* SD카드 Power On */
293 SD_PowerOn();
294
295 /* SPI 통신을 위해 Chip Select */
296 SELECT();
297
298 /* SD카드 타입변수 초기화 */
299 type = 0;
300
301 /* Idle 상태 진입 */
302 if (SD_SendCmd(CMD0, 0) == 1)
303 {
304 /* 타이머 1초 설정 */
305 Timer1 = 100;
306
307 /* SD 인터페이스 동작 조건 확인 */
308 if (SD_SendCmd(CMD8, 0x1AA) == 1)
309 {
310 /* SDC Ver2+ */
311 for (n = 0; n < 4; n++)
312 {
313 ocr[n] = SPI_RxByte();
314 }
315
316 if (ocr[2] == 0x01 && ocr[3] == 0xAA)
317 {
318 /* 2.7-3.6V 전압범위 동작 */
319 do {
320 if (SD_SendCmd(CMD55, 0) <= 1 && SD_SendCmd(CMD41, 1UL << 30) == 0)
321 break; /* ACMD41 with HCS bit */
322 } while (Timer1);
323
324 if (Timer1 && SD_SendCmd(CMD58, 0) == 0)
325 {
326 /* Check CCS bit */
327 for (n = 0; n < 4; n++)
328 {
329 ocr[n] = SPI_RxByte();
330 }
331
332 type = (ocr[0] & 0x40) ? 6 : 2;
333 }
334 }
335 }
336 else
337 {
338 /* SDC Ver1 or MMC */
339 type = (SD_SendCmd(CMD55, 0) <= 1 && SD_SendCmd(CMD41, 0) <= 1) ? 2 : 1; /* SDC : MMC */
340
341 do {
342 if (type == 2)
343 {
344 if (SD_SendCmd(CMD55, 0) <= 1 && SD_SendCmd(CMD41, 0) == 0)
345 break; /* ACMD41 */
346 }
347 else
348 {
349 if (SD_SendCmd(CMD1, 0) == 0)
350 break; /* CMD1 */
351 }
352 } while (Timer1);
353
354 if (!Timer1 || SD_SendCmd(CMD16, 512) != 0)
355 {
356 /* 블럭 길이 선택 */
357 type = 0;
358 }
359 }
360 }
361
362 CardType = type;
363
364 DESELECT();
365
366 SPI_RxByte(); /* Idle 상태 전환 (Release DO) */
367
368 if (type)
369 {
370 /* Clear STA_NOINIT */
371 Stat &= ~STA_NOINIT;
372 }
373 else
374 {
375 /* Initialization failed */
376 SD_PowerOff();
377 }
378
379 return Stat;
380}

391/* 섹터 읽기 */
392DRESULT SD_disk_read(BYTE pdrv, BYTE* buff, DWORD sector, UINT count)
393{
394 if (pdrv || !count)
395 return RES_PARERR;
396
397 if (Stat & STA_NOINIT)
398 return RES_NOTRDY;
399
400 if (!(CardType & 4))
401 sector *= 512; /* 지정 sector를 Byte addressing 단위로 변경 */
402
403 SELECT();
404
405 if (count == 1)
406 {
407 /* 싱글 블록 읽기 */
408 if ((SD_SendCmd(CMD17, sector) == 0) && SD_RxDataBlock(buff, 512))
409 count = 0;
410 }
411 else
412 {
413 /* 다중 블록 읽기 */
414 if (SD_SendCmd(CMD18, sector) == 0)
415 {
416 do {
417 if (!SD_RxDataBlock(buff, 512))
418 break;
419
420 buff += 512;
421 } while (--count);
422
423 /* STOP_TRANSMISSION, 모든 블럭을 다 읽은 후, 전송 중지 요청 */
424 SD_SendCmd(CMD12, 0);
425 }
426 }
427
428 DESELECT();
429 SPI_RxByte(); /* Idle 상태(Release DO) */
430
431 return count ? RES_ERROR : RES_OK;
432}

490/* 기타 함수 */
491DRESULT SD_disk_ioctl(BYTE drv, BYTE ctrl, void *buff)
492{
493 DRESULT res;
494 BYTE n, csd[16], *ptr = buff;
495 WORD csize;
496
497 if (drv)
498 return RES_PARERR;
499
500 res = RES_ERROR;
501
502 if (ctrl == CTRL_POWER)
503 {
504 switch (*ptr)
505 {
506 case 0:
507 if (SD_CheckPower())
508 SD_PowerOff(); /* Power Off */
509 res = RES_OK;
510 break;
511 case 1:
512 SD_PowerOn(); /* Power On */
513 res = RES_OK;
514 break;
515 case 2:
516 *(ptr + 1) = (BYTE) SD_CheckPower();
517 res = RES_OK; /* Power Check */
518 break;
519 default:
520 res = RES_PARERR;
521 }
522 }
523 else
524 {
525 if (Stat & STA_NOINIT)
526 return RES_NOTRDY;
527
528 SELECT();
529
530 switch (ctrl)
531 {
532 case GET_SECTOR_COUNT:
533 /* SD카드 내 Sector의 개수 (DWORD) */
534 if ((SD_SendCmd(CMD9, 0) == 0) && SD_RxDataBlock(csd, 16))
535 {
536 if ((csd[0] >> 6) == 1)
537 {
538 /* SDC ver 2.00 */
539 csize = csd[9] + ((WORD) csd[8] << 8) + 1;
540 *(DWORD*) buff = (DWORD) csize << 10;
541 }
542 else
543 {
544 /* MMC or SDC ver 1.XX */
545 n = (csd[5] & 15) + ((csd[10] & 128) >> 7) + ((csd[9] & 3) << 1) + 2;
546 csize = (csd[8] >> 6) + ((WORD) csd[7] << 2) + ((WORD) (csd[6] & 3) << 10) + 1;
547 *(DWORD*) buff = (DWORD) csize << (n - 9);
548 }
549
550 res = RES_OK;
551 }
552 break;
553
554 case GET_SECTOR_SIZE:
555 /* 섹터의 단위 크기 (WORD) */
556 *(WORD*) buff = 512;
557 res = RES_OK;
558 break;
559
560 case CTRL_SYNC:
561 /* 쓰기 동기화 */
562 if (SD_ReadyWait() == 0xFF)
563 res = RES_OK;
564 break;
565
566 case MMC_GET_CSD:
567 /* CSD 정보 수신 (16 bytes) */
568 if (SD_SendCmd(CMD9, 0) == 0 && SD_RxDataBlock(ptr, 16))
569 res = RES_OK;
570 break;
571
572 case MMC_GET_CID:
573 /* CID 정보 수신 (16 bytes) */
574 if (SD_SendCmd(CMD10, 0) == 0 && SD_RxDataBlock(ptr, 16))
575 res = RES_OK;
576 break;
577
578 case MMC_GET_OCR:
579 /* OCR 정보 수신 (4 bytes) */
580 if (SD_SendCmd(CMD58, 0) == 0)
581 {
582 for (n = 0; n < 4; n++)
583 {
584 *ptr++ = SPI_RxByte();
585 }
586
587 res = RES_OK;
588 }
589
590 default:
591 res = RES_PARERR;
592 }
593
594 DESELECT();
595 SPI_RxByte();
596 }
597
598 return res;
599}

326Filter_Status ProcessFilter(struct BQFilter *F, float *data){
327
328 float input = 0;
329 float out = 0;
330
331 input = *data;
332
333 out = (F->b0 / F->a0) * input + (F->b1 / F->a0) * F->z[0] + (F->b2 / F->a0) * F->z[1] - (F->a1 / F->a0) * F->z[2] - (F->a2 / F->a0) * F->z[3];
334
335 F->z[3] = F->z[2];
336 F->z[2] = out;
337 F->z[1] = F->z[0];
338 F->z[0] = input;
339 *data = out;
340
341 return FILTER_OK;
342}

379Filter_Status SetupLowpass(struct BQFilter *LP, float cutoff, float Q){
380
381 //float A = 1;
382 float w = 2 * M_PI * cutoff / samplerate;
383
384 float sin = sinf(w);
385 float cos = cosf(w);
386
387 float alpha = sin / (2 * Q);
388
389 LP->b0 = (1 - cos) / 2;
390 LP->b1 = 1 - cos;
391 LP->b2 = LP->b0;
392 LP->a0 = 1 + alpha;
393 LP->a1 = -2 * cos;
394 LP->a2 = 1 - alpha;
395
396 return FILTER_OK;

552Filter_Status SetupLowShelf(struct BQFilter *LS, float cutoff, float Q, float dBGain){
553
554 float exp = (dBGain) / 40;
555 float A = powf(10, exp);
556 float w = 2 * M_PI * cutoff / samplerate;
557
558 float sin = sinf(w);
559 float cos = cosf(w);
560
561 float beta = sqrtf(A)/Q;
562
563 LS->b0 = A * ((A + 1) - (A - 1) * cos + beta * sin);
564 LS->b1 = 2 * A * ((A - 1) - (A + 1) * cos);
565 LS->b2 = A * ((A + 1) - (A - 1) * cos - beta * sin);
566 LS->a0 = (A + 1) + (A - 1) * cos + beta * sin;
567 LS->a1 = -2 * ((A - 1) + (A + 1) * cos);
568 LS->a2 = (A + 1) + (A - 1) * cos - beta * sin;
569
570 return FILTER_OK;
571}

583Filter_Status SetupHighShelf(struct BQFilter *HS, float cutoff, float Q, float dBGain){
584
585 float exp = (dBGain) / 40;
586 float A = powf(10, exp);
587 float w = 2 * M_PI * cutoff / samplerate;
588
589 float sin = sinf(w);
590 float cos = cosf(w);
591
592 float beta = sqrtf(A)/Q;
593
594 HS->b0 = A * ((A + 1) + (A - 1) * cos + beta * sin);
595 HS->b1 = -2 * A * ((A - 1) + (A + 1) * cos);
596 HS->b2 = A * ((A + 1) + (A - 1) * cos - beta * sin);
597 HS->a0 = (A + 1) - (A - 1) * cos + beta * sin;
598 HS->a1 = 2 * ((A - 1) - (A + 1) * cos);
599 HS->a2 = (A + 1) - (A - 1) * cos - beta * sin;
